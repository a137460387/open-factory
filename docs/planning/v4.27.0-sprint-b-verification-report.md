## 六、使用示例

### 智能导出调度

```typescript
import { scheduleExport, getRecommendedExportConfig } from '@open-factory/editor-core';

// 获取推荐配置
const config = getRecommendedExportConfig(project);

// 执行智能调度
const decision = scheduleExport(exportPlan, project, config);

console.log(`推荐 preset: ${decision.preset}`);
console.log(`推荐线程数: ${decision.threads}`);
console.log(`推荐 CRF: ${decision.crf}`);
console.log(`预计速度: ${decision.estimatedSpeedMultiplier}x`);
console.log(`预计文件大小: ${decision.estimatedFileSizeMb}MB`);
```

### 导出预览画布

```tsx
import { ExportPreviewCanvas } from './components/export/ExportPreviewCanvas';

function ExportProgress({ taskId }) {
  return (
    <ExportPreviewCanvas
      taskId={taskId}
      width={640}
      height={360}
      maxFrames={9}
      showProgressBar={true}
      showTimestamps={true}
    />
  );
}
```

### VMAF 质量监控

```typescript
import { 
  detectVmafCapabilities, 
  generateVmafReport,
  createDegradedQualityReport 
} from '@open-factory/editor-core';

// 检测环境能力
const capabilities = await detectVmafCapabilities();

if (capabilities.vmafAvailable) {
  // 生成质量报告
  const report = generateVmafReport(result, project.name);
  console.log(report);
} else {
  // 降级方案
  const degradedReport = createDegradedQualityReport(plan, settings, duration);
  console.log(`估算质量分数: ${degradedReport.averageVmaf}`);
}
```
